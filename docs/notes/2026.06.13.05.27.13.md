## `shape/sphere.clj`

```clojure
(ns shape.sphere
  "Pure coordinate morphisms and geometry for a spherical body.
   No I/O. No domain semantics. All angles in radians, distances in
   whatever unit the caller passes (typically km or planet-radii)."
  (:require [clojure.math :as math]))

;; ── Coordinate Transforms ─────────────────────────────────────────────────────

(defn sph->cart
  "Spherical [φ θ r] → Cartesian [x y z].
   φ = latitude  (-π/2 .. π/2)
   θ = longitude ( 0   .. 2π )
   r = radius"
  [[φ θ r]]
  [(* r (math/cos φ) (math/cos θ))
   (* r (math/cos φ) (math/sin θ))
   (* r (math/sin φ))])

(defn cart->sph
  "Cartesian [x y z] → Spherical [φ θ r].
   Inverse of sph->cart."
  [[x y z]]
  (let [r (math/sqrt (+ (* x x) (* y y) (* z z)))]
    [(math/asin  (/ z r))
     (math/atan2 y x)
     r]))

;; ── Vector Primitives ─────────────────────────────────────────────────────────

(defn vec-dot    [[ax ay az] [bx by bz]] (+ (* ax bx) (* ay by) (* az bz)))
(defn vec-scale  [s [x y z]]             [(* s x) (* s y) (* s z)])
(defn vec-add    [[ax ay az] [bx by bz]] [(+ ax bx) (+ ay by) (+ az bz)])
(defn vec-sub    [[ax ay az] [bx by bz]] [(- ax bx) (- ay by) (- az bz)])
(defn vec-magnitude [[x y z]]            (math/sqrt (+ (* x x) (* y y) (* z z))))

(defn vec-normalize
  "Unit vector. Returns nil for zero vector to avoid NaN propagation."
  [v]
  (let [m (vec-magnitude v)]
    (when (> m 0.0)
      (vec-scale (/ 1.0 m) v))))

;; ── Surface Operations ────────────────────────────────────────────────────────

(defn surface-normal
  "Outward unit normal at spherical surface point [φ θ].
   On a unit sphere this equals the Cartesian position of the point."
  [[φ θ]]
  (vec-normalize (sph->cart [φ θ 1.0])))

(defn angular-distance
  "Great-circle angular distance between two surface points [φ1 θ1] [φ2 θ2].
   Uses the haversine formula for numerical stability near 0 and π."
  [[φ1 θ1] [φ2 θ2]]
  (let [dφ (- φ2 φ1)
        dθ (- θ2 θ1)
        a  (+ (math/pow (math/sin (/ dφ 2)) 2)
              (* (math/cos φ1)
                 (math/cos φ2)
                 (math/pow (math/sin (/ dθ 2)) 2)))]
    (* 2 (math/asin (math/sqrt a)))))

(defn surface-distance
  "Great-circle distance in the same units as planet-radius."
  [p1 p2 planet-radius]
  (* planet-radius (angular-distance p1 p2)))

;; ── Solar Geometry ────────────────────────────────────────────────────────────

(defn solar-elevation
  "Sine of the solar elevation angle at a surface point.
   sun-direction: unit Cartesian vector pointing from planet centre toward sun.
   surface-point: [φ θ] on the planet surface.
   Returns value in [-1.0 .. 1.0]:
     > 0  → sun above horizon (day)
     = 0  → exactly on terminator
     < 0  → sun below horizon (night)"
  [sun-direction [φ θ]]
  (vec-dot sun-direction (surface-normal [φ θ])))

;; ── Ray-Sphere Intersection ───────────────────────────────────────────────────

(defn ray-intersect
  "Intersect a ray with a sphere of `radius` centred at the origin.
   ray: {:origin [x y z] :dir [dx dy dz]}  (dir need not be normalized)
   Returns the Cartesian hit point closest to the origin (i.e. the entry
   point), or nil if the ray misses or originates inside the sphere."
  [{:keys [origin dir]} radius]
  (let [d   (vec-normalize dir)
        oc  origin
        b   (vec-dot d oc)
        c   (- (vec-dot oc oc) (* radius radius))
        dis (- (* b b) c)]
    (when (> dis 0.0)
      (let [t1 (- (- b) (math/sqrt dis))
            t2 (+ (- b) (math/sqrt dis))]
        (when (> t1 0.0)            ; entry point in front of ray origin
          (vec-add origin (vec-scale t1 d)))))))
```

## `shape/grid.clj`

The icosphere is the correct geodesic grid for a planet — no pole singularity, uniform cell area, six-connected (mostly). The base icosphere has 12 vertices and 20 faces. Subdivision 4 gives you 2562 cells (~15,000 km² each for an Earth-sized planet). Subdivision 6 gives you 40,962 cells (~1,000 km² each).

```clojure
(ns shape.grid
  "Icosphere geodesic grid.
   Provides: cell adjacency, point-to-cell lookup, centroid, solid angle.
   Pure shape morphisms — no domain semantics."
  (:require [clojure.math :as math]
            [shape.sphere :as sph]))

;; ── Base Icosahedron ──────────────────────────────────────────────────────────

(def ^:private φ-gold (/ (+ 1.0 (math/sqrt 5.0)) 2.0))

(defn- normalize-v [[x y z]]
  (let [m (math/sqrt (+ (* x x) (* y y) (* z z)))]
    [(/ x m) (/ y m) (/ z m)]))

(defn base-icosphere
  "Construct the base icosahedron projected onto the unit sphere.
   Returns {:vertices [...] :faces [...] :edges #{...} :cells {...}}
   where cells are keyed by face index."
  []
  (let [verts (mapv normalize-v
                    [[0  1  φ-gold] [0 -1  φ-gold] [0  1 (- φ-gold)] [0 -1 (- φ-gold)]
                     [ 1  φ-gold 0] [-1  φ-gold 0] [ 1 (- φ-gold) 0] [-1 (- φ-gold) 0]
                     [ φ-gold 0  1] [(- φ-gold) 0  1] [ φ-gold 0 -1] [(- φ-gold) 0 -1]])
        faces [[0 1 8] [0 8 4] [0 4 5] [0 5 9] [0 9 1]
               [1 6 8] [8 6 10] [8 10 4] [4 10 2] [4 2 5]
               [5 2 11] [5 11 9] [9 11 7] [9 7 1] [1 7 6]
               [3 6 7] [3 7 11] [3 11 2] [3 2 10] [3 10 6]]
        edges (into #{} (mapcat (fn [[a b c]] [#{a b} #{b c} #{a c}]) faces))
        cells (into {} (map-indexed (fn [i f] [i {:face f :level 0}]) faces))]
    {:vertices verts :faces faces :edges edges :cells cells}))

;; ── Subdivision ───────────────────────────────────────────────────────────────

(defn- midpoint-key [a b] (if (< a b) [a b] [b a]))

(defn- subdivide-once
  "One step of Loop subdivision: split each triangle into four by
   inserting midpoints on each edge, projected back onto the unit sphere."
  [{:keys [vertices faces]}]
  (let [mid-cache (atom {})
        verts     (atom (vec vertices))
        get-mid   (fn [a b]
                    (let [k (midpoint-key a b)]
                      (or (@mid-cache k)
                          (let [m  (normalize-v
                                    (mapv #(/ (+ %1 %2) 2.0)
                                          (nth @verts a)
                                          (nth @verts b)))
                                idx (count @verts)]
                            (swap! verts conj m)
                            (swap! mid-cache assoc k idx)
                            idx))))]
    (let [new-faces (into []
                          (mapcat (fn [[a b c]]
                                    (let [ab (get-mid a b)
                                          bc (get-mid b c)
                                          ac (get-mid a c)]
                                      [[a ab ac]
                                       [b bc ab]
                                       [c ac bc]
                                       [ab bc ac]]))
                                  faces))]
      {:vertices @verts
       :faces    new-faces
       :edges    (into #{} (mapcat (fn [[a b c]] [#{a b} #{b c} #{a c}]) new-faces))
       :cells    (into {} (map-indexed (fn [i f] [i {:face f}]) new-faces))})))

(defn subdivide
  "Subdivide the icosphere n times. Each step quadruples the face count."
  [ico n]
  (nth (iterate subdivide-once ico) n))

;; ── Cell Geometry ─────────────────────────────────────────────────────────────

(defn cell-centroid-cart
  "Cartesian centroid of a triangular cell, projected onto the unit sphere."
  [{:keys [vertices]} cell-id]
  (let [[a b c] (get-in (if (map? cell-id)
                           cell-id
                           {:face (get-in {} [cell-id :face])})
                         [:face])
        [a b c] (-> {} :faces (nth 0))]
    ;; resolve via cells map
    nil))

;; Rewrite using direct cells map access:
(defn- face-of [grid cell-id]
  (get-in grid [:cells cell-id :face]))

(defn cell->centroid
  "Surface [φ θ] centroid of cell-id."
  [grid cell-id]
  (let [[a b c] (face-of grid cell-id)
        verts   (:vertices grid)
        [ax ay az] (nth verts a)
        [bx by bz] (nth verts b)
        [cx cy cz] (nth verts c)
        cx'     (/ (+ ax bx cx) 3.0)
        cy'     (/ (+ ay by cy) 3.0)
        cz'     (/ (+ az bz cz) 3.0)]
    (-> [cx' cy' cz'] normalize-v sph/cart->sph (subvec 0 2))))

(defn cell-solid-angle
  "Approximate solid angle (steradians) subtended by cell-id.
   Uses the spherical excess formula for the spherical triangle."
  [grid cell-id]
  (let [[a b c] (face-of grid cell-id)
        verts   (:vertices grid)
        va (nth verts a)
        vb (nth verts b)
        vc (nth verts c)
        ;; Spherical excess = A + B + C - π
        ;; Angle at vertex A = arccos((vb·vc - (va·vb)(va·vc)) / ...)
        dot-fn  sph/vec-dot
        angle   (fn [p q r]
                  (let [pq (sph/vec-sub q p)
                        pr (sph/vec-sub r p)
                        pq-n (sph/vec-normalize pq)
                        pr-n (sph/vec-normalize pr)]
                    (math/acos (max -1.0 (min 1.0 (dot-fn pq-n pr-n))))))]
    (- (+ (angle va vb vc)
          (angle vb va vc)
          (angle vc va vb))
       Math/PI)))

;; ── Adjacency ─────────────────────────────────────────────────────────────────

(defn- build-adjacency
  "Precompute a map of cell-id → #{neighbor-cell-id} by shared edges."
  [{:keys [cells] :as grid}]
  (let [edge->cells (reduce-kv
                     (fn [acc cid {:keys [face]}]
                       (let [[a b c] face]
                         (-> acc
                             (update #{a b} (fnil conj #{}) cid)
                             (update #{b c} (fnil conj #{}) cid)
                             (update #{a c} (fnil conj #{}) cid))))
                     {}
                     cells)]
    (reduce-kv
     (fn [adj _edge cell-set]
       (reduce (fn [a cid]
                 (update a cid into (disj cell-set cid)))
               adj
               cell-set))
     (into {} (map (fn [k] [k #{}]) (keys cells)))
     edge->cells)))

(defn with-adjacency
  "Attach a :adjacency map to the grid. Compute once, reuse."
  [grid]
  (assoc grid :adjacency (build-adjacency grid)))

(defn neighbors
  "Return the set of cell IDs adjacent to cell-id.
   Requires grid built with with-adjacency."
  [grid cell-id]
  (get (:adjacency grid) cell-id #{}))

;; ── Point-to-Cell Lookup ──────────────────────────────────────────────────────

(defn- point-in-triangle?
  "Test if a Cartesian unit-sphere point p lies within the spherical
   triangle defined by vertices [a b c] using barycentric signs."
  [p a b c]
  (let [cross (fn [[ax ay az] [bx by bz]]
                [(- (* ay bz) (* az by))
                 (- (* az bx) (* ax bz))
                 (- (* ax by) (* ay bx))])
        dot   sph/vec-dot
        s1 (dot (cross a b) p)
        s2 (dot (cross b c) p)
        s3 (dot (cross c a) p)]
    (or (and (>= s1 0) (>= s2 0) (>= s3 0))
        (and (<= s1 0) (<= s2 0) (<= s3 0)))))

(defn sph->cell
  "Map a spherical surface point [φ θ] to its containing cell-id.
   O(F) scan — for hot paths, use a pre-built spatial index."
  [{:keys [vertices cells] :as _grid} [φ θ]]
  (let [p (sph/sph->cart [φ θ 1.0])]
    (reduce-kv
     (fn [_ cid {:keys [face]}]
       (let [[a b c] face
             va (nth vertices a)
             vb (nth vertices b)
             vc (nth vertices c)]
         (when (point-in-triangle? p va vb vc)
           (reduced cid))))
     nil
     cells)))
```

  "Surface point: [latitude longitude] in radians."

   [:vertices [:vector CartPoint]]

    (throw (ex-info "Invalid spherical point" {:point p :schema SphPoint}))))
